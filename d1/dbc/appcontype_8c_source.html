102 S_CIP_ConnectionObject *
103 getIOConnectionForConnectionData(S_CIP_ConnectionObject *pa_pstConnData,
104  EIP_UINT16 *pa_pnExtendedError)
105 {
106  S_CIP_ConnectionObject *pstRetVal = NULL;
107  *pa_pnExtendedError = 0;
108 
109  pstRetVal = getExclusiveOwnerConnection(pa_pstConnData, pa_pnExtendedError);
110  if (NULL == pstRetVal)
111  {
112  if (0 == *pa_pnExtendedError)
113  {
114  /* we found no connection and don't have an error so try input only next */
115  pstRetVal = getInputOnlyConnection(pa_pstConnData,
116  pa_pnExtendedError);
117  if (NULL == pstRetVal)
118  {
119  if (0 == *pa_pnExtendedError)
120  {
121  /* we found no connection and don't have an error so try listen only next */
122  pstRetVal = getListenOnlyConnection(pa_pstConnData,
123  pa_pnExtendedError);
124  if ((NULL == pstRetVal) && (0 == *pa_pnExtendedError))
125  {
126  /* no application connection type was found that suits the given data */
127  /* TODO check error code VS */
128  *pa_pnExtendedError =
129  CIP_CON_MGR_INCONSISTENT_APPLICATION_PATH_COMBO;
130  }
131  else
132  {
133  pa_pstConnData->m_eInstanceType = enConnTypeIOListenOnly;
134  }
135  }
136  }
137  else
138  {
139  pa_pstConnData->m_eInstanceType = enConnTypeIOInputOnly;
140  }
141  }
142  }
143  else
144  {
145  pa_pstConnData->m_eInstanceType = enConnTypeIOExclusiveOwner;
146  }
147 
148  if (NULL != pstRetVal)
149  {
150  copyConnectionData(pstRetVal, pa_pstConnData);
151  }
152 
153  return pstRetVal;
154 }

156 S_CIP_ConnectionObject *
157 getExclusiveOwnerConnection(S_CIP_ConnectionObject * pa_pstConnData,
158  EIP_UINT16 *pa_pnExtendedError)
159 {
160  S_CIP_ConnectionObject *pstRetVal = NULL;
161  int i;
162 
163  for (i = 0; i < OPENER_CIP_NUM_EXLUSIVE_OWNER_CONNS; i++)
164  {
165  if ((g_astExlusiveOwnerConnections[i].m_unOutputAssembly
166  == pa_pstConnData->ConnectionPath.ConnectionPoint[0])
167  && (g_astExlusiveOwnerConnections[i].m_unInputAssembly
168  == pa_pstConnData->ConnectionPath.ConnectionPoint[1])
169  && (g_astExlusiveOwnerConnections[i].m_unConfigAssembly
170  == pa_pstConnData->ConnectionPath.ConnectionPoint[2]))
171  {
172 
173  /* check if on other connection point with the same output assembly is currently connected */
174  if (NULL
175  != getConnectedOutputAssembly(
176  pa_pstConnData->ConnectionPath.ConnectionPoint[0]))
177  {
178  *pa_pnExtendedError = CIP_CON_MGR_ERROR_OWNERSHIP_CONFLICT;
179  break;
180  }
181  pstRetVal = &(g_astExlusiveOwnerConnections[i].m_stConnectionData);
182  break;
183  }
184  }
185  return pstRetVal;
186 }

188 S_CIP_ConnectionObject *
189 getInputOnlyConnection(S_CIP_ConnectionObject * pa_pstConnData,
190  EIP_UINT16 *pa_pnExtendedError)
191 {
192  S_CIP_ConnectionObject *pstRetVal = NULL;
193  int i, j;
194 
195  for (i = 0; i < OPENER_CIP_NUM_INPUT_ONLY_CONNS; i++)
196  {
197  if (g_astInputOnlyConnections[i].m_unOutputAssembly
198  == pa_pstConnData->ConnectionPath.ConnectionPoint[0])
199  { /* we have the same output assembly */
200  if (g_astInputOnlyConnections[i].m_unInputAssembly
201  != pa_pstConnData->ConnectionPath.ConnectionPoint[1])
202  {
203  *pa_pnExtendedError =
204  CIP_CON_MGR_INVALID_PRODUCING_APPLICATION_PATH;
205  break;
206  }
207  if (g_astInputOnlyConnections[i].m_unConfigAssembly
208  != pa_pstConnData->ConnectionPath.ConnectionPoint[2])
209  {
210  *pa_pnExtendedError =
211  CIP_CON_MGR_INCONSISTENT_APPLICATION_PATH_COMBO;
212  break;
213  }
214 
215  for (j = 0; j < OPENER_CIP_NUM_INPUT_ONLY_CONNS_PER_CON_PATH; j++)
216  {
217  if (CONN_STATE_NONEXISTENT
218  == g_astInputOnlyConnections[i].m_astConnectionData[j].State)
219  {
220  return &(g_astInputOnlyConnections[i].m_astConnectionData[j]);
221  }
222  }
223  *pa_pnExtendedError = CIP_CON_MGR_TARGET_OBJECT_OUT_OF_CONNECTIONS;
224  break;
225  }
226  }
227  return pstRetVal;

230 S_CIP_ConnectionObject *
231 getListenOnlyConnection(S_CIP_ConnectionObject * pa_pstConnData,
232  EIP_UINT16 *pa_pnExtendedError)
233 {
234  S_CIP_ConnectionObject *pstRetVal = NULL;
235  int i, j;
236 
237  if (CIP_MULTICAST_CONNECTION
238  != (pa_pstConnData->T_to_O_NetworkConnectionParameter
239  & CIP_MULTICAST_CONNECTION))
240  {
241  /* a listen only connection has to be a multicast connection. */
242  *pa_pnExtendedError = CIP_CON_MGR_NON_LISTEN_ONLY_CONNECTION_NOT_OPENED; /* maybe not the best error message however there is no suitable definition in the cip spec */
243  return NULL;
244  }
245 
246  for (i = 0; i < OPENER_CIP_NUM_LISTEN_ONLY_CONNS; i++)
247  {
248  if (g_astListenOnlyConnections[i].m_unOutputAssembly
249  == pa_pstConnData->ConnectionPath.ConnectionPoint[0])
250  { /* we have the same output assembly */
251  if (g_astListenOnlyConnections[i].m_unInputAssembly
252  != pa_pstConnData->ConnectionPath.ConnectionPoint[1])
253  {
254  *pa_pnExtendedError =
255  CIP_CON_MGR_INVALID_PRODUCING_APPLICATION_PATH;
256  break;
257  }
258  if (g_astListenOnlyConnections[i].m_unConfigAssembly
259  != pa_pstConnData->ConnectionPath.ConnectionPoint[2])
260  {
261  *pa_pnExtendedError =
262  CIP_CON_MGR_INCONSISTENT_APPLICATION_PATH_COMBO;
263  break;
264  }
265 
266  if (NULL
267  == getExistingProdMulticastConnection(
268  pa_pstConnData->ConnectionPath.ConnectionPoint[1]))
269  {
270  *pa_pnExtendedError =
271  CIP_CON_MGR_NON_LISTEN_ONLY_CONNECTION_NOT_OPENED;
272  break;
273  }
274 
275  for (j = 0; j < OPENER_CIP_NUM_LISTEN_ONLY_CONNS_PER_CON_PATH; j++)
276  {
277  if (CONN_STATE_NONEXISTENT
278  == g_astListenOnlyConnections[i].m_astConnectionData[j].State)
279  {
280  return &(g_astListenOnlyConnections[i].m_astConnectionData[j]);
281  }
282  }
283  *pa_pnExtendedError = CIP_CON_MGR_TARGET_OBJECT_OUT_OF_CONNECTIONS;
284  break;
285  }
286  }
287  return pstRetVal;
288 }

290 S_CIP_ConnectionObject *
291 getExistingProdMulticastConnection(EIP_UINT32 pa_unInputPoint)
292 {
293  S_CIP_ConnectionObject *pstRunner = g_pstActiveConnectionList;
294 
295  while (NULL != pstRunner)
296  {
297  if ((enConnTypeIOExclusiveOwner == pstRunner->m_eInstanceType)
298  || (enConnTypeIOInputOnly == pstRunner->m_eInstanceType))
299  {
300  if ((pa_unInputPoint == pstRunner->ConnectionPath.ConnectionPoint[1])
301  && (CIP_MULTICAST_CONNECTION
302  == (pstRunner->T_to_O_NetworkConnectionParameter
303  & CIP_MULTICAST_CONNECTION))
304  && (EIP_INVALID_SOCKET != pstRunner->sockfd[PRODUCING]))
305  {
306  /* we have a connection that produces the same input assembly,
307  * is a multicast producer and manages the connection.
308  */
309  break;
310  }
311  }
312  pstRunner = pstRunner->m_pstNext;
313  }
314  return pstRunner;
