126 registerClass(S_CIP_Class * pa_pt2Class)
127 {
128  S_CIP_MR_Object **p = &g_pt2firstObject;
129 
130  while (*p)
131  p = &(*p)->next; /* follow the list until p points to an empty link (list end)*/
132 
133  *p = (S_CIP_MR_Object *) IApp_CipCalloc(1, sizeof(S_CIP_MR_Object)); /* create a new node at the end of the list*/
134  if (*p == 0)
135  return EIP_ERROR; /* check for memory error*/
136 
137  (*p)->pt2Class = pa_pt2Class; /* fill in the new node*/
138  (*p)->next = NULL;
139 
140  return EIP_OK;

143 EIP_STATUS
144 notifyMR(EIP_UINT8 * pa_pnData, int pa_nDataLength)
145 {
146  EIP_STATUS nRetVal = EIP_OK_SEND;
147  EIP_BYTE nStatus;
148 
149  gMRResponse.Data = g_acMessageDataReplyBuffer; /* set reply buffer, using a fixed buffer (about 100 bytes) */
150 
151  OPENER_TRACE_INFO("notifyMR: routing unconnected message\n");
152  if (CIP_ERROR_SUCCESS
153  != (nStatus = createMRRequeststructure(pa_pnData, pa_nDataLength,
154  &gMRRequest)))
155  { /* error from create MR structure*/
156  OPENER_TRACE_ERR("notifyMR: error from createMRRequeststructure\n");
157  gMRResponse.GeneralStatus = nStatus;
158  gMRResponse.SizeofAdditionalStatus = 0;
159  gMRResponse.Reserved = 0;
160  gMRResponse.DataLength = 0;
161  gMRResponse.ReplyService = (0x80 | gMRRequest.Service);
162  }
163  else
164  {
165  /* forward request to appropriate Object if it is registered*/
166  S_CIP_MR_Object *pt2regObject;
167 
168  pt2regObject = getRegisteredObject(gMRRequest.RequestPath.ClassID);
169  if (pt2regObject == 0)
170  {
171  OPENER_TRACE_ERR(
172  "notifyMR: sending CIP_ERROR_OBJECT_DOES_NOT_EXIST reply, class id 0x%x is not registered\n",
173  (unsigned) gMRRequest.RequestPath.ClassID);
174  gMRResponse.GeneralStatus = CIP_ERROR_PATH_DESTINATION_UNKNOWN; /*according to the test tool this should be the correct error flag instead of CIP_ERROR_OBJECT_DOES_NOT_EXIST;*/
175  gMRResponse.SizeofAdditionalStatus = 0;
176  gMRResponse.Reserved = 0;
177  gMRResponse.DataLength = 0;
178  gMRResponse.ReplyService = (0x80 | gMRRequest.Service);
179  }
180  else
181  {
182  /* call notify function from Object with ClassID (gMRRequest.RequestPath.ClassID)
183  object will or will not make an reply into gMRResponse*/
184  gMRResponse.Reserved = 0;
185  OPENER_ASSERT(NULL != pt2regObject->pt2Class);
186  OPENER_TRACE_INFO(
187  "notifyMR: calling notify function of class '%s'\n",
188  pt2regObject->pt2Class->acName);
189  nRetVal = notifyClass(pt2regObject->pt2Class, &gMRRequest,
190  &gMRResponse);
191 
192 #ifdef OPENER_TRACE_ENABLED
193  if (nRetVal == EIP_ERROR)
194  {
195  OPENER_TRACE_ERR(
196  "notifyMR: notify function of class '%s' returned an error\n",
197  pt2regObject->pt2Class->acName);
198  }
199  else if (nRetVal == EIP_OK)
200  {
201  OPENER_TRACE_INFO(
202  "notifyMR: notify function of class '%s' returned no reply\n",
203  pt2regObject->pt2Class->acName);
204  }
205  else
206  {
207  OPENER_TRACE_INFO(
208  "notifyMR: notify function of class '%s' returned a reply\n",
209  pt2regObject->pt2Class->acName);
210  }
211 #endif
212  }
213  }
214  return nRetVal;
215 }
