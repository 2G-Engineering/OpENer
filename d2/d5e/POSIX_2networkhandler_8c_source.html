212 NetworkHandler_ProcessOnce(void)
213 {
214  int fd;
215  int res;
216 
217  read_fds = master;
218 
219  tv.tv_sec = 0;
220  tv.tv_usec = (
221  TheNetworkStatus.elapsedtime < OPENER_TIMER_TICK ? OPENER_TIMER_TICK
222  - TheNetworkStatus.elapsedtime :
223  0) * 1000; /* 10 ms */
224 
225  res = select(fdmax + 1, &read_fds, 0, 0, &tv);
226 
227  if (res == -1)
228  {
229  if (EINTR == errno) /* we have somehow been interrupted. The default behavior is to go back into the select loop. */
230  {
231  return EIP_OK;
232  }
233  else
234  {
235  OPENER_TRACE_ERR("networkhandler: error with select: %s\n", strerror(errno));
236  return EIP_ERROR;
237  }
238  }
239 
240  if (res > 0)
241  {
242 
243  checkAndHandleTCPListenerSocket();
244  checkAndHandleUDPBroadCastSocket();
245  checkAndHandleConsumingUDPSockets();
246 
247  for (fd = 0; fd <= fdmax; fd++)
248  {
249  if (true == checkSocketSet(fd))
250  {
251  /* if it is still checked it is a TCP receive */
252  if (EIP_ERROR == handleDataOnTCPSocket(fd)) /* if error */
253  {
254  IApp_CloseSocket(fd);
255  closeSession(fd); /* clean up session and close the socket */
256  }
257  }
258  }
259  }
260 
261  actualtime = getmilliseconds();
262  TheNetworkStatus.elapsedtime += actualtime - lasttime;
263  lasttime = actualtime;
264 
265  /* check if we had been not able to update the connection manager for several OPENER_TIMER_TICK.
266  * This should compensate the jitter of the windows timer
267  */
268  while (TheNetworkStatus.elapsedtime >= OPENER_TIMER_TICK)
269  {
270  /* call manage_connections() in connection manager every OPENER_TIMER_TICK ms */
271  manageConnections();
272  TheNetworkStatus.elapsedtime -= OPENER_TIMER_TICK;
273  }
274  return EIP_OK;
275 }

332  EIP_UINT8 *rxp;
333  long nCheckVal;
334  size_t unDataSize;
335  long nDataSent;
336  int nRemainingBytes = 0;
337 
338  /* We will handle just one EIP packet here the rest is done by the select
339  * method which will inform us if more data is available in the socket
340  because of the current implementation of the main loop this may not be
341  the fastest way and a loop here with a non blocking socket would better
342  fit*/
343 
344  /*Check how many data is here -- read the first four bytes from the connection */
345  nCheckVal = recv(pa_nSocket, g_acPCEthernetCommBuffer, 4, 0); /*TODO we may have to set the socket to a non blocking socket */
346 
347  if (nCheckVal == 0)
348  {
349  OPENER_TRACE_ERR("networkhandler: connection closed by client: %s\n", strerror(errno));
350  return EIP_ERROR;
351  }
352  if (nCheckVal < 0)
353  {
354  OPENER_TRACE_ERR("networkhandler: error on recv: %s\n", strerror(errno));
355  return EIP_ERROR;
356  }
357 
358  rxp = &g_acPCEthernetCommBuffer[2]; /* at this place EIP stores the data length */
359  unDataSize = ltohs(&rxp) + ENCAPSULATION_HEADER_LENGTH - 4; /* -4 is for the 4 bytes we have already read*/
360  /* (NOTE this advances the buffer pointer) */
361  if (PC_OPENER_ETHERNET_BUFFER_SIZE - 4 < unDataSize)
362  { /*TODO can this be handled in a better way?*/
363  OPENER_TRACE_ERR("too large packet received will be ignored, will drop the data\n");
364  /* Currently we will drop the whole packet */
365  nDataSent = PC_OPENER_ETHERNET_BUFFER_SIZE;
366 
367  do
368  {
369  nCheckVal = recv(pa_nSocket, g_acPCEthernetCommBuffer, nDataSent, 0);
370 
371  if (nCheckVal == 0) /* got error or connection closed by client */
372  {
373  OPENER_TRACE_ERR("networkhandler: connection closed by client: %s\n", strerror(errno));
374  return EIP_ERROR;
375  }
376  if (nCheckVal < 0)
377  {
378  OPENER_TRACE_ERR("networkhandler: error on recv: %s\n", strerror(errno));
379  return EIP_ERROR;
380  }
381  unDataSize -= nCheckVal;
382  if ((unDataSize < PC_OPENER_ETHERNET_BUFFER_SIZE)
383  && (unDataSize != 0))
384  {
385  nDataSent = unDataSize;
386  }
387  }
388  while (0 != unDataSize); /*TODO fragile end statement */
389  return EIP_OK;
390  }
391 
392  nCheckVal = recv(pa_nSocket, &g_acPCEthernetCommBuffer[4], unDataSize, 0);
393 
394  if (nCheckVal == 0) /* got error or connection closed by client */
395  {
396  OPENER_TRACE_ERR("networkhandler: connection closed by client: %s\n", strerror(errno));
397  return EIP_ERROR;
398  }
399  if (nCheckVal < 0)
400  {
401  OPENER_TRACE_ERR("networkhandler: error on recv: %s\n", strerror(errno));
402  return EIP_ERROR;
403  }
404 
405  if ((unsigned) nCheckVal == unDataSize)
406  {
407  /*we got the right amount of data */
408  unDataSize += 4;
409  /*TODO handle partial packets*/
410  OPENER_TRACE_INFO("Data received on tcp:\n");
411 
412  g_nCurrentActiveTCPSocket = pa_nSocket;
413 
414  nCheckVal = handleReceivedExplictTCPData(pa_nSocket,
415  g_acPCEthernetCommBuffer, unDataSize, &nRemainingBytes);
416 
417  g_nCurrentActiveTCPSocket = -1;
418 
419  if (nRemainingBytes != 0)
420  {
421  OPENER_TRACE_WARN("Warning: received packet was to long: %d Bytes left!\n",
422  nRemainingBytes);
423  }
424 
425  if (nCheckVal > 0)
426  {
427  OPENER_TRACE_INFO("reply sent:\n");
428 
429  nDataSent = send(pa_nSocket, (char *) g_acPCEthernetCommBuffer,
430  nCheckVal, 0);
431  if (nDataSent != nCheckVal)
432  {
433  OPENER_TRACE_WARN("TCP response was not fully sent\n");
434  }
435  }
436 
437  return EIP_OK;
438  }
439  else
440  {
441  /* we got a fragmented packet currently we cannot handle this will
442  * for this we would need a network buffer per TCP socket
443  *
444  * However with typical packet sizes of EIP this should't be a big issue.
445  */
446  /*TODO handle fragmented packets */
447  }
448  return EIP_ERROR;

454 IApp_CreateUDPSocket(int pa_nDirection, struct sockaddr_in *pa_pstAddr)
455 {

461  socklen_t nPeerAddrLen;
462 #endif
463 
464  nPeerAddrLen = sizeof(struct sockaddr_in);
465  /* create a new UDP socket */
466  if ((newfd = socket(PF_INET, SOCK_DGRAM, 0)) == -1)
467  {
468  OPENER_TRACE_ERR("networkhandler: cannot create UDP socket: %s\n", strerror(errno));
469  return EIP_INVALID_SOCKET;
470  }
471 
472  OPENER_TRACE_INFO("networkhandler: UDP socket %d\n", newfd);
473 
474  /* check if it is sending or receiving */
475  if (pa_nDirection == CONSUMING)
476  {
477  int nOptVal = 1;
478  if (setsockopt(newfd, SOL_SOCKET, SO_REUSEADDR, (char *)&nOptVal, sizeof(nOptVal))
479  == -1)
480  {
481  OPENER_TRACE_ERR("error setting socket option SO_REUSEADDR on consuming udp socket\n");
482  return EIP_ERROR;
483  }
484

486  if ((bind(newfd, (struct sockaddr *) pa_pstAddr, sizeof(struct sockaddr)))
487  == -1)
488  {
489  OPENER_TRACE_ERR("error on bind udp: %s\n", strerror(errno));
490  return EIP_INVALID_SOCKET;
491  }
492 
493  OPENER_TRACE_INFO("networkhandler: bind UDP socket %d\n", newfd);
494  }
495  else
496  { /* we have a producing udp socket */
497 
498  if (pa_pstAddr->sin_addr.s_addr == g_stMultiCastconfig.m_unMcastStartAddr)
499  {
500  if (1 != g_unTTLValue)
501  { /* we need to set a TTL value for the socket */
502  if (setsockopt(newfd, IPPROTO_IP, IP_MULTICAST_TTL,
503  &g_unTTLValue, sizeof(g_unTTLValue) < 0))
504  {
505  OPENER_TRACE_ERR("networkhandler: could not set the TTL to: %d, error: %s\n", g_unTTLValue, strerror(errno));
506  return EIP_INVALID_SOCKET;
507  }
508  }
509  }
510  }
511 
512  if ((pa_nDirection == CONSUMING) || (0 == pa_pstAddr->sin_addr.s_addr))
513  {
514  /* we have a peer to peer producer or a consuming connection*/
515  if (getpeername(g_nCurrentActiveTCPSocket, (struct sockaddr *) &stPeerAdr,
516  &nPeerAddrLen) < 0)
517  {
518  OPENER_TRACE_ERR("networkhandler: could not get peername: %s\n", strerror(errno));
519  return EIP_INVALID_SOCKET;
520  }
521  /* store the originators address */
522  pa_pstAddr->sin_addr.s_addr = stPeerAdr.sin_addr.s_addr;
523  }
524 
525  /* add new fd to the master list */
526  FD_SET(newfd, &master);
527  if (newfd > fdmax)
528  {
529  fdmax = newfd;
530  }
531  return newfd;
532 }

594 {

607  if (true == checkSocketSet(TheNetworkStatus.nUDPListener))
608  {
609

611 
612  OPENER_TRACE_STATE(
613  "networkhandler: unsolicited UDP message on EIP broadcast socket\n");
614 
615  /*Handle udp broadcast messages */
616  nReceived_size = recvfrom(TheNetworkStatus.nUDPListener,
617  g_acPCEthernetCommBuffer, PC_OPENER_ETHERNET_BUFFER_SIZE, 0,
618  (struct sockaddr *) &stFrom, &nFromLen);
619 
620  if (nReceived_size <= 0)
621  { /* got error */
622  OPENER_TRACE_ERR(
623  "networkhandler: error on recvfrom udp broadcast port: %s\n",
624  strerror(errno));
625  return;
626  }
627 
628  OPENER_TRACE_INFO("Data received on udp:\n");
629 
630  rxp = &g_acPCEthernetCommBuffer[0];
631  do
632  {
633  nReplyLen = handleReceivedExplictUDPData(
634  TheNetworkStatus.nUDPListener, &stFrom, rxp, nReceived_size,
635  &nRemainingBytes);
636 
637  rxp += nReceived_size - nRemainingBytes;
638  nReceived_size = nRemainingBytes;
639 
640  if (nReplyLen > 0)
641  {
642  OPENER_TRACE_INFO("reply sent:\n");
643 
644  /* if the active fd matches a registered UDP callback, handle a UDP packet */
645  if (sendto(TheNetworkStatus.nUDPListener,
646  (char *) g_acPCEthernetCommBuffer, nReplyLen, 0,
647  (struct sockaddr *) &stFrom, sizeof(stFrom)) != nReplyLen)
648  {
649  OPENER_TRACE_INFO(

651  }
652  }
653  }
654  while (nRemainingBytes > 0);
