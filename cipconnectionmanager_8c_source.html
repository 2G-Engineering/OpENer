241

412  CIP_ERROR_SUCCESS, 0);

545 EIP_STATUS
546 manageConnections(void)
547 {
548  EIP_STATUS res;
549  S_CIP_ConnectionObject *pstRunner;
550 
551  /*Inform application that it can execute */
552  IApp_HandleApplication();
553  manageEncapsulationMessages();
554 
555  pstRunner = g_pstActiveConnectionList;
556  while (NULL != pstRunner)
557  {
558  if (pstRunner->State == CONN_STATE_ESTABLISHED)
559  {
560  if ((0 != pstRunner->p_stConsumingInstance) || /* we have a consuming connection check innacitivitywatchdog timer */
561  (pstRunner->TransportTypeClassTrigger & 0x80)) /* all sever connections have to maintain an innacitivitywatchdog timer */
562  {
563  pstRunner->InnacitvityWatchdogTimer -= OPENER_TIMER_TICK;
564  if (pstRunner->InnacitvityWatchdogTimer <= 0)
565  {
566  /* we have a timed out connection perform watchdog time out action*/
567  OPENER_TRACE_INFO(">>>>>>>>>>Connection timed out\n");OPENER_ASSERT(
568  NULL != pstRunner->m_pfTimeOutFunc);
569  pstRunner->m_pfTimeOutFunc(pstRunner);
570  }
571  }
572  /* only if the connection has not timed out check if data is to be send */
573  if (CONN_STATE_ESTABLISHED == pstRunner->State)
574  {
575  /* client connection */
576  if ((pstRunner->ExpectedPacketRate != 0)
577  && (EIP_INVALID_SOCKET != pstRunner->sockfd[PRODUCING])) /* only produce for the master connection */
578  {
579  if (CIP_CONN_CYCLIC_CONNECTION
580  != (pstRunner->TransportTypeClassTrigger
581  & CIP_CONN_PRODUCTION_TRIGGER_MASK))
582  {
583  /* non cyclic connections have to decrement production inhibit timer */
584  if (0 <= pstRunner->m_nProductionInhibitTimer)
585  {
586  pstRunner->m_nProductionInhibitTimer -=
587  OPENER_TIMER_TICK;
588  }
589  }
590  pstRunner->TransmissionTriggerTimer -= OPENER_TIMER_TICK;
591  if (pstRunner->TransmissionTriggerTimer <= 0)
592  { /* need to send package */
593  OPENER_ASSERT(NULL != pstRunner->m_pfSendDataFunc);
594  res = pstRunner->m_pfSendDataFunc(pstRunner);
595  if (res == EIP_ERROR)
596  {
597  OPENER_TRACE_ERR(
598  "sending of UDP data in manage Connection failed\n");
599  }
600  /* reload the timer value */
601  pstRunner->TransmissionTriggerTimer =
602  pstRunner->ExpectedPacketRate;
603  if (CIP_CONN_CYCLIC_CONNECTION
604  != (pstRunner->TransportTypeClassTrigger
605  & CIP_CONN_PRODUCTION_TRIGGER_MASK))
606  {
607  /* non cyclic connections have to reload the production inhibit timer */
608  pstRunner->m_nProductionInhibitTimer =
609  pstRunner->m_unProductionInhibitTime;
610  }
611  }

851 checkElectronicKeyData(EIP_UINT8 pa_nKeyFormat, S_CIP_KeyData *pa_pstKeyData,
852  EIP_UINT16 *pa_pnExtStatus)
853 {
854  EIP_BYTE nCompatiblityMode = pa_pstKeyData->MajorRevision & 0x80;
855 
856  /* Remove compatibility bit */
857  pa_pstKeyData->MajorRevision &= 0x7F;
858 
859  /* Default return value */
860  *pa_pnExtStatus = CIP_CON_MGR_SUCCESS;
861 
862  /* Check key format */
863  if (4 != pa_nKeyFormat)
864  {
865  *pa_pnExtStatus = CIP_CON_MGR_ERROR_INVALID_SEGMENT_TYPE_IN_PATH;
866  return EIP_ERROR;
867  }
868 
869  /* Check VendorID and ProductCode, must match, or 0 */
870  if (((pa_pstKeyData->VendorID != VendorID) && (pa_pstKeyData->VendorID != 0)) ||
871  ((pa_pstKeyData->ProductCode != ProductCode) && (pa_pstKeyData->ProductCode != 0)))

876  else
877  {
878  /* VendorID and ProductCode are correct */
879 
880  /* Check DeviceType, must match or 0 */

882  {
883  *pa_pnExtStatus = CIP_CON_MGR_ERROR_DEVICE_TYPE_ERROR;
884  return EIP_ERROR;
885  }
886  else
887  {
888  /* VendorID, ProductCode and DeviceType are correct */
889 
890  if (!nCompatiblityMode)
891  {
892  /* Major = 0 is valid */
893  if (0 == pa_pstKeyData->MajorRevision)
894  {
895  return(EIP_OK);
896  }
897 
898  /* Check Major / Minor Revision, Major must match, Minor match or 0 */
899  if ((pa_pstKeyData->MajorRevision != Revison.MajorRevision) ||
900  ((pa_pstKeyData->MinorRevision != Revison.MinorRevision) && (pa_pstKeyData->MinorRevision != 0)))
901  {
902  *pa_pnExtStatus = CIP_CON_MGR_ERROR_REVISION_MISMATCH;
903  return EIP_ERROR;
904  }
905  }
906  else
907  {
908  /* Compatiblitiy mode is set */
909 
910  /* Major must macth, Minor != 0 and <= MinorRevision */
911  if ((pa_pstKeyData->MajorRevision == Revison.MajorRevision) &&
912  (pa_pstKeyData->MinorRevision > 0) &&
913  (pa_pstKeyData->MinorRevision <= Revison.MinorRevision))
914  {
915  return(EIP_OK);
916  }
917  else
918  {
919  *pa_pnExtStatus = CIP_CON_MGR_ERROR_REVISION_MISMATCH;
920  return EIP_ERROR;
921  }
922  } /* end if CompatiblityMode handling */
923  }
924  }
925 
926  return (*pa_pnExtStatus == CIP_CON_MGR_SUCCESS) ? EIP_OK : EIP_ERROR;

929 EIP_UINT8
930 parseConnectionPath(S_CIP_ConnectionObject *pa_pstConnObj,
931  S_CIP_MR_Request *pa_MRRequest, EIP_UINT16 *pa_pnExtendedError)
932 {
933  int i;
934  EIP_UINT8 *pnMsg = pa_MRRequest->Data;
935  int nRemainingPathSize = pa_pstConnObj->ConnectionPathSize = *pnMsg++; /* length in words */
936  S_CIP_Class *pstClass = NULL;
937  int O2TConnectionType, T2OConnectionType, imax;
938 
939  /* with 256 we mark that we haven't got a PIT segment */
940  pa_pstConnObj->m_unProductionInhibitTime = 256;
941 
942  if ((FORWARD_OPEN_HEADER_LENGTH + nRemainingPathSize * 2)
943  < pa_MRRequest->DataLength)
944  {
945  /* the received packet is larger than the data in the path */
946  *pa_pnExtendedError = 0;
947  return CIP_ERROR_TOO_MUCH_DATA;
948  }
949 
950  if ((FORWARD_OPEN_HEADER_LENGTH + nRemainingPathSize * 2)
951  > pa_MRRequest->DataLength)
952  {
953  /*there is not enough data in received packet */
954  *pa_pnExtendedError = 0;
955  return CIP_ERROR_NOT_ENOUGH_DATA;
956  }
957 
958  if (nRemainingPathSize > 0)
959  {
960  /* first electronic key */
961  if (*pnMsg == 0x34)
962  {
963  if (nRemainingPathSize < 5)
964  {
965  /*there is not enough data for holding the electronic key segement*/
966  *pa_pnExtendedError = 0;
967  return CIP_ERROR_NOT_ENOUGH_DATA;
968  }
969 
970  /* logical electronic key found */
971  pa_pstConnObj->ElectronicKey.SegmentType = 0x34;
972  pnMsg++;
973  pa_pstConnObj->ElectronicKey.KeyFormat = *pnMsg++;
974  pa_pstConnObj->ElectronicKey.KeyData.VendorID = ltohs(&pnMsg);
975  pa_pstConnObj->ElectronicKey.KeyData.DeviceType = ltohs(&pnMsg);
976  pa_pstConnObj->ElectronicKey.KeyData.ProductCode = ltohs(&pnMsg);
977  pa_pstConnObj->ElectronicKey.KeyData.MajorRevision = *pnMsg++;
978  pa_pstConnObj->ElectronicKey.KeyData.MinorRevision = *pnMsg++;
979  nRemainingPathSize -= 5; /*length of the electronic key*/
980  OPENER_TRACE_INFO(
981  "key: ven ID %d, dev type %d, prod code %d, major %d, minor %d\n",
982  pa_pstConnObj->ElectronicKey.KeyData.VendorID,
983  pa_pstConnObj->ElectronicKey.KeyData.DeviceType,
984  pa_pstConnObj->ElectronicKey.KeyData.ProductCode,
985  pa_pstConnObj->ElectronicKey.KeyData.MajorRevision,
986  pa_pstConnObj->ElectronicKey.KeyData.MinorRevision);
987 
988  if (EIP_OK
989  != checkElectronicKeyData(pa_pstConnObj->ElectronicKey.KeyFormat,
990  &(pa_pstConnObj->ElectronicKey.KeyData), pa_pnExtendedError))
991  {
992  return CIP_ERROR_CONNECTION_FAILURE;
993  }
994  }
995  else
996  {
997  OPENER_TRACE_INFO("no key\n");
998  }
999 
1000  if (CIP_CONN_CYCLIC_CONNECTION
1001  != (pa_pstConnObj->TransportTypeClassTrigger
1002  & CIP_CONN_PRODUCTION_TRIGGER_MASK))
1003  {
1004  /*non cyclic connections may have a production inhibit */
1005  if (CIP_CONN_PIT_NETWORK_SEGEMENT_ID == *pnMsg)
1006  {
1007  pa_pstConnObj->m_unProductionInhibitTime = pnMsg[1];
1008  pnMsg += 2;
1009  nRemainingPathSize -= 1;
1010  }
1011  }
1012 
1013  if (EQLOGICALPATH(*pnMsg,0x20))
1014  { /* classID */
1015  pa_pstConnObj->ConnectionPath.ClassID = getPaddedLogicalPath(&pnMsg);
1016  pstClass = getCIPClass(pa_pstConnObj->ConnectionPath.ClassID);
1017  if (0 == pstClass)
1018  {
1019  OPENER_TRACE_ERR("classid %"PRIx32" not found\n",
1020  pa_pstConnObj->ConnectionPath.ClassID);
1021  if (pa_pstConnObj->ConnectionPath.ClassID >= 0xC8) /*reserved range of class ids */
1022 
1023  {
1024  *pa_pnExtendedError =
1025  CIP_CON_MGR_ERROR_INVALID_SEGMENT_TYPE_IN_PATH;
1026  }
1027  else
1028  {
1029  *pa_pnExtendedError =
1030  CIP_CON_MGR_INCONSISTENT_APPLICATION_PATH_COMBO;
1031  }
1032  return CIP_ERROR_CONNECTION_FAILURE;
1033  }OPENER_TRACE_INFO("classid %"PRIx32" (%s)\n",
1034  pa_pstConnObj->ConnectionPath.ClassID, pstClass->acName);
1035  }
1036  else
1037  {
1038  *pa_pnExtendedError = CIP_CON_MGR_ERROR_INVALID_SEGMENT_TYPE_IN_PATH;
1039  return CIP_ERROR_CONNECTION_FAILURE;
1040  }
1041  nRemainingPathSize -= 1; /* 1 16Bit word for the class part of the path */
1042 
1043  if (EQLOGICALPATH(*pnMsg,0x24))
1044  { /* store the configuration ID for later checking in the application connection types */
1045  pa_pstConnObj->ConnectionPath.ConnectionPoint[2] =
1046  getPaddedLogicalPath(&pnMsg);
1047  OPENER_TRACE_INFO("Configuration instance id %"PRId32"\n", pa_pstConnObj->ConnectionPath.ConnectionPoint[2]);
1048  if (NULL
1049  == getCIPInstance(pstClass,
1050  pa_pstConnObj->ConnectionPath.ConnectionPoint[2]))
1051  {
1052  /*according to the test tool we should respond with this extended error code */
1053  *pa_pnExtendedError =
1054  CIP_CON_MGR_ERROR_INVALID_SEGMENT_TYPE_IN_PATH;
1055  return CIP_ERROR_CONNECTION_FAILURE;
1056  }
1057  /* 1 or 2 16Bit words for the configuration instance part of the path */
1058  nRemainingPathSize -=
1059  (pa_pstConnObj->ConnectionPath.ConnectionPoint[2] > 0xFF) ? 2 : 1;
1060  }
1061  else
1062  {
1063  OPENER_TRACE_INFO("no config data\n");
1064  }
1065 
1066  if (0x03 == (pa_pstConnObj->TransportTypeClassTrigger & 0x03))
1067  {
1068  /*we have Class 3 connection*/
1069  if (nRemainingPathSize > 0)
1070  {
1071  OPENER_TRACE_WARN(
1072  "Too much data in connection path for class 3 connection\n");
1073  *pa_pnExtendedError =
1074  CIP_CON_MGR_ERROR_INVALID_SEGMENT_TYPE_IN_PATH;
1075  return CIP_ERROR_CONNECTION_FAILURE;
1076  }
1077 
1078  /* connection end point has to be the message router instance 1 */
1079  if ((pa_pstConnObj->ConnectionPath.ClassID
1080  != CIP_MESSAGE_ROUTER_CLASS_CODE)
1081  || (pa_pstConnObj->ConnectionPath.ConnectionPoint[2] != 1))
1082  {
1083  *pa_pnExtendedError = CIP_CON_MGR_INCONSISTENT_APPLICATION_PATH_COMBO;
1084  return CIP_ERROR_CONNECTION_FAILURE;
1085  }
1086  pa_pstConnObj->ConnectionPath.ConnectionPoint[0] =
1087  pa_pstConnObj->ConnectionPath.ConnectionPoint[2];
1088  }
1089  else
1090  { /* we have an IO connection */
1091  O2TConnectionType = (pa_pstConnObj->O_to_T_NetworkConnectionParameter
1092  & 0x6000) >> 13;
1093  T2OConnectionType = (pa_pstConnObj->T_to_O_NetworkConnectionParameter
1094  & 0x6000) >> 13;
1095 
1096  pa_pstConnObj->ConnectionPath.ConnectionPoint[1] = 0; /* set not available path to Invalid */
1097 
1098  if (O2TConnectionType == 0)
1099  {
1100  if (T2OConnectionType == 0)
1101  { /* configuration only connection */
1102  imax = 0;
1103  OPENER_TRACE_WARN("assembly: type invalid\n");
1104  }
1105  else
1106  { /* 1 path -> path is for production */
1107  OPENER_TRACE_INFO("assembly: type produce\n");
1108  imax = 1;
1109  }
1110  }
1111  else
1112  {
1113  if (T2OConnectionType == 0)
1114  { /* 1 path -> path is for consumption */
1115  OPENER_TRACE_INFO("assembly: type consume\n");
1116  imax = 1;
1117  }
1118  else
1119  { /* 2 pathes -> 1st for production 2nd for consumption */
1120  OPENER_TRACE_INFO("assembly: type bidirectional\n");
1121  imax = 2;
1122  }
1123  }
1124 
1125  for (i = 0; i < imax; i++) /* process up to 2 encoded paths */
1126  {
1127  if (EQLOGICALPATH(*pnMsg,0x24) || EQLOGICALPATH(*pnMsg,0x2C)) /* Connection Point interpreted as InstanceNr -> only in Assembly Objects */
1128  { /* InstanceNR */
1129  pa_pstConnObj->ConnectionPath.ConnectionPoint[i] =
1130  getPaddedLogicalPath(&pnMsg);
1131  OPENER_TRACE_INFO("connection point %"PRIu32"\n",
1132  pa_pstConnObj->ConnectionPath.ConnectionPoint[i]);
1133  if (0
1134  == getCIPInstance(pstClass,
1135  pa_pstConnObj->ConnectionPath.ConnectionPoint[i]))
1136  {
1137  *pa_pnExtendedError =
1138  CIP_CON_MGR_INCONSISTENT_APPLICATION_PATH_COMBO;
1139  return CIP_ERROR_CONNECTION_FAILURE;
1140  }
1141  /* 1 or 2 16Bit word for the connection point part of the path */
1142  nRemainingPathSize -=
1143  (pa_pstConnObj->ConnectionPath.ConnectionPoint[i] > 0xFF) ? 2 :
1144  1;
1145  }
1146  else
1147  {
1148  *pa_pnExtendedError =
1149  CIP_CON_MGR_ERROR_INVALID_SEGMENT_TYPE_IN_PATH;
1150  return CIP_ERROR_CONNECTION_FAILURE;
1151  }
1152  }
1153 
1154  g_unConfigDataLen = 0;
1155  g_pnConfigDataBuffer = NULL;
1156 
1157  while (nRemainingPathSize > 0)
1158  { /* have something left in the path should be configuration data */
1159 
1160  switch (*pnMsg)
1161  {
1162  case 0x80:
1163  /* we have a simple data segment */
1164  g_unConfigDataLen = pnMsg[1] * 2; /*data segments store length 16-bit word wise */
1165  g_pnConfigDataBuffer = &(pnMsg[2]);
1166  nRemainingPathSize -= (g_unConfigDataLen + 2);
1167  pnMsg += (g_unConfigDataLen + 2);
1168  break;
1169  /*TODO do we have to handle ANSI extended symbol data segments too? */
1170  case CIP_CONN_PIT_NETWORK_SEGEMENT_ID:
1171  if (CIP_CONN_CYCLIC_CONNECTION
1172  != (pa_pstConnObj->TransportTypeClassTrigger
1173  & CIP_CONN_PRODUCTION_TRIGGER_MASK))
1174  {
1175  /* only non cyclic connections may have a production inhibit */
1176  pa_pstConnObj->m_unProductionInhibitTime = pnMsg[1];
1177  pnMsg += 2;
1178  nRemainingPathSize -= 2;
1179  }
1180  else
1181  {
1182  *pa_pnExtendedError = pa_pstConnObj->ConnectionPathSize
1183  - nRemainingPathSize; /*offset in 16Bit words where within the connection path the error happend*/
1184  return 0x04; /*status code for invalid segment type*/
1185  }
1186  break;
1187  default:
1188  OPENER_TRACE_WARN(
1189  "No data segment identifier found for the configuration data\n");
1190  *pa_pnExtendedError = pa_pstConnObj->ConnectionPathSize
1191  - nRemainingPathSize; /*offset in 16Bit words where within the connection path the error happend*/
1192  return 0x04; /*status code for invalid segment type*/
1193  break;
1194  }
1195  }
1196  }
1197  }
1198 
1199  /*save back the current position in the stream allowing followers to parse anything thats still there*/
1200  pa_MRRequest->Data = pnMsg;
1201  return EIP_OK;
1202 }
