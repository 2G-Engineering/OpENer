64 configureNetworkInterface(const char *pa_acIpAdress,
65  const char *pa_acSubNetMask, const char *pa_acGateway)
66 {
67  EIP_UINT32 nHostId;
68 
69  Interface_Configuration.IPAddress = inet_addr(pa_acIpAdress);
70  Interface_Configuration.NetworkMask = inet_addr(pa_acSubNetMask);
71  Interface_Configuration.Gateway = inet_addr(pa_acGateway);
72 
73  /* calculate the CIP multicast address. The multicast address is calculated, not input*/
74  nHostId = ntohl(Interface_Configuration.IPAddress)
75  & ~ntohl(Interface_Configuration.NetworkMask); /* see CIP spec 3-5.3 for multicast address algorithm*/
76  nHostId -= 1;
77  nHostId &= 0x3ff;
78 
79  g_stMultiCastconfig.m_unMcastStartAddr = htonl(
80  ntohl(inet_addr("239.192.1.0")) + (nHostId << 5));
81 
82  return EIP_OK;
83 }

219 getAttributeSingleTCPIPInterface(S_CIP_Instance *pa_pstInstance,
220  S_CIP_MR_Request *pa_pstMRRequest, S_CIP_MR_Response *pa_pstMRResponse)
221 {
222  EIP_STATUS nRetVal = EIP_OK_SEND;
223  EIP_BYTE *paMsg = pa_pstMRResponse->Data;
224  EIP_UINT32 unMultiCastAddr;
225 
226  if (9 == pa_pstMRRequest->RequestPath.AttributNr)
227  { /* attribute 9 can not be easily handled with the default mechanism therefore we will do it by hand */
228  pa_pstMRResponse->DataLength = 0;
229  pa_pstMRResponse->ReplyService = (0x80 | pa_pstMRRequest->Service);
230  pa_pstMRResponse->GeneralStatus = CIP_ERROR_SUCCESS;
231  pa_pstMRResponse->SizeofAdditionalStatus = 0;
232 
233  pa_pstMRResponse->DataLength += encodeData(CIP_USINT,
234  &(g_stMultiCastconfig.m_unAllocControl), &paMsg);
235  pa_pstMRResponse->DataLength += encodeData(CIP_USINT,
236  &(g_stMultiCastconfig.m_unReserved), &paMsg);
237  pa_pstMRResponse->DataLength += encodeData(CIP_UINT,
238  &(g_stMultiCastconfig.m_unNumMcast), &paMsg);
239 
240  unMultiCastAddr = ntohl(
241  g_stMultiCastconfig.m_unMcastStartAddr);
242 
243  pa_pstMRResponse->DataLength += encodeData(CIP_UDINT, &unMultiCastAddr,
244  &paMsg);
245  }
246  else
247  {
248  nRetVal = getAttributeSingle(pa_pstInstance, pa_pstMRRequest,
249  pa_pstMRResponse);
250  }
251  return nRetVal;
252 }

254 EIP_STATUS
255 getAttributeAllTCPIPInterface(S_CIP_Instance * pa_pstInstance,
256  S_CIP_MR_Request * pa_stMRRequest, S_CIP_MR_Response * pa_stMRResponse)
257 {
258 
259  int j;
260  EIP_UINT8 *ptmp;
261  int nAttrNum;
262  S_CIP_attribute_struct *pstAttribute;
263 
264  ptmp = pa_stMRResponse->Data; /* pointer into the reply */
265  pstAttribute = pa_pstInstance->pstAttributes;
266 
267  for (j = 0; j < pa_pstInstance->pstClass->nNr_of_Attributes; j++) /* for each instance attribute of this class */
268  {
269  nAttrNum = pstAttribute->CIP_AttributNr;
270  if (nAttrNum < 32
271  && (pa_pstInstance->pstClass->nGetAttrAllMask & 1 << nAttrNum)) /* only return attributes that are flagged as being part of GetAttributeALl */
272  {
273  pa_stMRRequest->RequestPath.AttributNr = nAttrNum;
274 
275  if (8 == nAttrNum)
276  { /* insert 6 zeros for the required empty safety network number according to Table 5-3.10 */
277  memset(pa_stMRResponse->Data, 0, 6);
278  pa_stMRResponse->Data += 6;
279  }
280 
281  if (EIP_OK_SEND
282  != getAttributeSingleTCPIPInterface(pa_pstInstance,
283  pa_stMRRequest, pa_stMRResponse))
284  {
285  pa_stMRResponse->Data = ptmp;
286  return EIP_ERROR;
287  }
288  pa_stMRResponse->Data += pa_stMRResponse->DataLength;
289  }
290  pstAttribute++;
291  }
292  pa_stMRResponse->DataLength = pa_stMRResponse->Data - ptmp;
293  pa_stMRResponse->Data = ptmp;
294 
295  return EIP_OK_SEND;
296 }
