125 S_CIP_Instance *
126 addCIPInstances(S_CIP_Class * pa_pstCIPClass, int pa_nNr_of_Instances)
127 {
128  S_CIP_Instance *first, *p, **pp;
129  int i;
130  int inst = 1; /* the first instance is number 1 */
131 
132  OPENER_TRACE_INFO("adding %d instances to class %s\n", pa_nNr_of_Instances,
133  pa_pstCIPClass->acName);
134 
135  pp = &pa_pstCIPClass->pstInstances; /* get address of pointer to head of chain */
136  while (*pp) /* as long as what pp points to is not zero */
137  {
138  pp = &(*pp)->pstNext; /* follow the chain until pp points to pointer that contains a zero */
139  inst++; /* keep track of what the first new instance number will be */
140  }
141 
142  first = p = (S_CIP_Instance *) IApp_CipCalloc(pa_nNr_of_Instances,
143  sizeof(S_CIP_Instance)); /* allocate a block of memory for all created instances*/
144  OPENER_ASSERT(NULL != p);
145  /* fail if run out of memory */
146 
147  pa_pstCIPClass->nNr_of_Instances += pa_nNr_of_Instances; /* add the number of instances just created to the total recorded by the class */
148 
149  for (i = 0; i < pa_nNr_of_Instances; i++) /* initialize all the new instances */
150  {
151  *pp = p; /* link the previous pointer to this new node */
152 
153  p->nInstanceNr = inst; /* assign the next sequential instance number */
154  p->pstClass = pa_pstCIPClass; /* point each instance to its class */
155 
156  if (pa_pstCIPClass->nNr_of_Attributes) /* if the class calls for instance attributes */
157  { /* then allocate storage for the attribute array */
158  p->pstAttributes = (S_CIP_attribute_struct*) IApp_CipCalloc(
159  pa_pstCIPClass->nNr_of_Attributes,
160  sizeof(S_CIP_attribute_struct));
161  }
162 
163  pp = &p->pstNext; /* update pp to point to the next link of the current node */
164  inst++; /* update to the number of the next node*/
165  p++; /* point to the next node in the calloc'ed array*/
166  }
167 
168  return first;
169 }

431 int
432 encodeData(EIP_UINT8 pa_nCIP_Type, void *pa_pt2data, EIP_UINT8 **pa_pnMsg)
433 {
434  int counter = 0;
435  S_CIP_Byte_Array *p;
436 
437  switch (pa_nCIP_Type)
438  /* check the datatype of attribute */
439  {
440  case (CIP_BOOL):
441  case (CIP_SINT):
442  case (CIP_USINT):
443  case (CIP_BYTE):
444  **pa_pnMsg = *(EIP_UINT8 *) (pa_pt2data);
445  ++(*pa_pnMsg);
446  counter = 1;
447  break;
448 
449  case (CIP_INT):
450  case (CIP_UINT):
451  case (CIP_WORD):
452  htols(*(EIP_UINT16 *) (pa_pt2data), pa_pnMsg);
453  counter = 2;
454  break;
455 
456  case (CIP_DINT):
457  case (CIP_UDINT):
458  case (CIP_DWORD):
459  case (CIP_REAL):
460  htoll(*(EIP_UINT32 *) (pa_pt2data), pa_pnMsg);
461  counter = 4;
462  break;
463 
464 #ifdef OPENER_SUPPORT_64BIT_DATATYPES
465  case (CIP_LINT):
466  case (CIP_ULINT):
467  case (CIP_LWORD):
468  case (CIP_LREAL):
469  htol64(*(EIP_UINT64 *) (pa_pt2data), pa_pnMsg);
470  counter = 8;
471  break;
472 #endif
473 
474  case (CIP_STIME):
475  case (CIP_DATE):
476  case (CIP_TIME_OF_DAY):
477  case (CIP_DATE_AND_TIME):
478  break;
479  case (CIP_STRING):
480  {
481  S_CIP_String *s = (S_CIP_String *) pa_pt2data;
482 
483  htols(*(EIP_UINT16 *) &(s->Length), pa_pnMsg);
484  memcpy(*pa_pnMsg, s->String, s->Length);
485  *pa_pnMsg += s->Length;
486 
487  counter = s->Length + 2; /* we have a two byte length field */
488  if (counter & 0x01)
489  {
490  /* we have an odd byte count */
491  **pa_pnMsg = 0;
492  ++(*pa_pnMsg);
493  counter++;
494  }
495  break;
496  }
497  case (CIP_STRING2):
498  case (CIP_FTIME):
499  case (CIP_LTIME):
500  case (CIP_ITIME):
501  case (CIP_STRINGN):
502  break;
503 
504  case (CIP_SHORT_STRING):
505  {
506  S_CIP_Short_String *ss = (S_CIP_Short_String *) pa_pt2data;
507 
508  **pa_pnMsg = ss->Length;
509  ++(*pa_pnMsg);
510 
511  memcpy(*pa_pnMsg, ss->String, ss->Length);
512  *pa_pnMsg += ss->Length;
513 
514  counter = ss->Length + 1;
515  break;
516  }
517 
518  case (CIP_TIME):
519  break;
520 
521  case (CIP_EPATH):
522  counter = encodeEPath((S_CIP_EPATH *) pa_pt2data, pa_pnMsg);
523  break;
524 
525  case (CIP_ENGUNIT):
526  break;
527 
528  case (CIP_USINT_USINT):
529  {
530  S_CIP_Revision *rv = (S_CIP_Revision *) pa_pt2data;
531 
532  **pa_pnMsg = rv->MajorRevision;
533  ++(*pa_pnMsg);
534  **pa_pnMsg = rv->MinorRevision;
535  ++(*pa_pnMsg);
536  counter = 2;
537  break;
538  }
539 
540  case (CIP_UDINT_UDINT_UDINT_UDINT_UDINT_STRING):
541  {
542  /* TCP/IP attribute 5 */

597 int
598 decodeData(EIP_UINT8 pa_nCIP_Type, void *pa_pt2data, EIP_UINT8 **pa_pnMsg)
599 {
600  int nRetVal = -1;
601 
602  switch (pa_nCIP_Type)
603  /* check the datatype of attribute */
604  {
605  case (CIP_BOOL):
606  case (CIP_SINT):
607  case (CIP_USINT):
608  case (CIP_BYTE):
609  *(EIP_UINT8 *) (pa_pt2data) = **pa_pnMsg;
610  ++(*pa_pnMsg);
611  nRetVal = 1;
612  break;
613 
614  case (CIP_INT):
615  case (CIP_UINT):
616  case (CIP_WORD):
617  (*(EIP_UINT16 *) (pa_pt2data)) = ltohs(pa_pnMsg);
618  nRetVal = 2;
619  break;
620 
621  case (CIP_DINT):
622  case (CIP_UDINT):
623  case (CIP_DWORD):
624  (*(EIP_UINT32 *) (pa_pt2data)) = ltohl(pa_pnMsg);
625  nRetVal = 4;
626  break;
627 
628 #ifdef OPENER_SUPPORT_64BIT_DATATYPES
629  case (CIP_LINT):
630  case (CIP_ULINT):
631  case (CIP_LWORD):
632  {
633  (*(EIP_UINT64 *) (pa_pt2data)) = ltoh64(pa_pnMsg);
634  nRetVal = 8;
635  }
636  break;
637 #endif
638 
639  case (CIP_STRING):
640  {
641  S_CIP_String *s = (S_CIP_String *) pa_pt2data;
642  s->Length = ltohs(pa_pnMsg);
643  memcpy(s->String, *pa_pnMsg, s->Length);
644  *pa_pnMsg += s->Length;
645 
646  nRetVal = s->Length + 2; /* we have a two byte length field */
647  if (nRetVal & 0x01)
648  {
649  /* we have an odd byte count */
650  ++(*pa_pnMsg);
651  nRetVal++;
652  }
653  }
654  break;
655  case (CIP_SHORT_STRING):
656  {
657  S_CIP_Short_String *ss = (S_CIP_Short_String *) pa_pt2data;
658 
659  ss->Length = **pa_pnMsg;
660  ++(*pa_pnMsg);
661 
662  memcpy(ss->String, *pa_pnMsg, ss->Length);
663  *pa_pnMsg += ss->Length;
664 
665  nRetVal = ss->Length + 1;
666  break;
667  }
668 
669  default:
670  break;
671  }
672 
673  return nRetVal;
674 }

676 EIP_STATUS
677 getAttributeAll(S_CIP_Instance * pa_pstInstance,
678  S_CIP_MR_Request * pa_stMRRequest, S_CIP_MR_Response * pa_stMRResponse)
679 {
680  int i, j;
681  EIP_UINT8 *ptmp;
682  S_CIP_attribute_struct *p_attr;
683  S_CIP_service_struct *p_service;
684 
685  ptmp = pa_stMRResponse->Data; /* pointer into the reply */
686  p_attr = pa_pstInstance->pstAttributes; /* pointer to list of attributes*/
687  p_service = pa_pstInstance->pstClass->pstServices; /* pointer to list of services*/
688 
689  if (pa_pstInstance->nInstanceNr == 2)
690  {
691  OPENER_TRACE_INFO("GetAttributeAll: instance number 2\n");
692  }
693 
694  for (i = 0; i < pa_pstInstance->pstClass->nNr_of_Services; i++) /* hunt for the GET_ATTRIBUTE_SINGLE service*/
695  {
696  if (p_service->CIP_ServiceNr == CIP_GET_ATTRIBUTE_SINGLE) /* found the service */
697  {
698  if (0 == pa_pstInstance->pstClass->nNr_of_Attributes)
699  {
700  pa_stMRResponse->DataLength = 0; /*there are no attributes to be sent back*/
701  pa_stMRResponse->ReplyService = (0x80 | pa_stMRRequest->Service);
702  pa_stMRResponse->GeneralStatus = CIP_ERROR_SERVICE_NOT_SUPPORTED;
703  pa_stMRResponse->SizeofAdditionalStatus = 0;
704  }
705  else
706  {
707  for (j = 0; j < pa_pstInstance->pstClass->nNr_of_Attributes; j++) /* for each instance attribute of this class */
708  {
709  int attrNum = p_attr->CIP_AttributNr;
710  if (attrNum < 32
711  && (pa_pstInstance->pstClass->nGetAttrAllMask
712  & 1 << attrNum)) /* only return attributes that are flagged as being part of GetAttributeALl */
713  {
714  pa_stMRRequest->RequestPath.AttributNr = attrNum;
715  if (EIP_OK_SEND
716  != p_service->m_ptfuncService(pa_pstInstance,
717  pa_stMRRequest, pa_stMRResponse))
718  {
719  pa_stMRResponse->Data = ptmp;
720  return EIP_ERROR;
721  }
722  pa_stMRResponse->Data += pa_stMRResponse->DataLength;
723  }
724  p_attr++;
725  }
726  pa_stMRResponse->DataLength = pa_stMRResponse->Data - ptmp;
727  pa_stMRResponse->Data = ptmp;
728  }
729  return EIP_OK_SEND;
730  }
731  p_service++;
732  }
733  return EIP_OK; /* Return 0 if cannot find GET_ATTRIBUTE_SINGLE service*/
734 }

736 int
737 encodeEPath(S_CIP_EPATH *pa_pstEPath, EIP_UINT8 **pa_pnMsg)
738 {
739  int nLen;
740 
741  nLen = pa_pstEPath->PathSize;
742  htols(pa_pstEPath->PathSize, pa_pnMsg);
743 
744  if (pa_pstEPath->ClassID < 256)
745  {
746  **pa_pnMsg = 0x20; /*8Bit Class Id */
747  ++(*pa_pnMsg);
748  **pa_pnMsg = (EIP_UINT8) pa_pstEPath->ClassID;
749  ++(*pa_pnMsg);
750  nLen -= 1;
751  }
752  else
753  {
754  **pa_pnMsg = 0x21; /*16Bit Class Id */
755  ++(*pa_pnMsg);
756  **pa_pnMsg = 0; /*padd byte */
757  ++(*pa_pnMsg);
758  htols(pa_pstEPath->ClassID, pa_pnMsg);
759  nLen -= 2;
760  }
761 
762  if (0 < nLen)
763  {
764  if (pa_pstEPath->InstanceNr < 256)
765  {
766  **pa_pnMsg = 0x24; /*8Bit Instance Id */
767  ++(*pa_pnMsg);
768  **pa_pnMsg = (EIP_UINT8) pa_pstEPath->InstanceNr;
769  ++(*pa_pnMsg);
770  nLen -= 1;
771  }
772  else
773  {
774  **pa_pnMsg = 0x25; /*16Bit Instance Id */
775  ++(*pa_pnMsg);
776  **pa_pnMsg = 0; /*padd byte */
777  ++(*pa_pnMsg);
778  htols(pa_pstEPath->InstanceNr, pa_pnMsg);
779  nLen -= 2;
780  }
781 
782  if (0 < nLen)
783  {
784  if (pa_pstEPath->AttributNr < 256)
785  {
786  **pa_pnMsg = 0x30; /*8Bit Attribute Id */
787  ++(*pa_pnMsg);
788  **pa_pnMsg = (EIP_UINT8) pa_pstEPath->AttributNr;
789  ++(*pa_pnMsg);
790  nLen -= 1;
791  }
792  else
793  {
794  **pa_pnMsg = 0x31; /*16Bit Attribute Id */
795  ++(*pa_pnMsg);
796  **pa_pnMsg = 0; /*padd byte */
797  ++(*pa_pnMsg);
798  htols(pa_pstEPath->AttributNr, pa_pnMsg);
799  nLen -= 2;
800  }
801  }
802  }
803 
804  return 2 + pa_pstEPath->PathSize * 2; /*path size is in 16 bit chunks according to the spec */
805 }

807 int
808 decodePaddedEPath(S_CIP_EPATH *pa_pstEPath, EIP_UINT8 ** pa_pnMsg)
809 {
810  int i;
811  EIP_UINT8 *pnRunner = *pa_pnMsg;
812 
813  pa_pstEPath->PathSize = *pnRunner;
814  pnRunner++;
815  /* copy path to structure, in version 0.1 only 8 bit for Class,Instance and Attribute, need to be replaced with function */
816  pa_pstEPath->ClassID = 0;
817  pa_pstEPath->InstanceNr = 0;
818  pa_pstEPath->AttributNr = 0;
819 
820  for (i = 0; i < pa_pstEPath->PathSize; i++)
821  {
822  if (0xE0 == ((*pnRunner) & 0xE0))
823  {
824  /*Invalid segment type*/
825  return EIP_ERROR;
826  }
827 
828  switch (*pnRunner)
829  {
830  case 0x20: /* classID */
831  pa_pstEPath->ClassID = *(EIP_UINT8 *) (pnRunner + 1);
832  pnRunner += 2;
833  break;
834 
835  case 0x21: /*classID 16Bit */
836  pnRunner += 2;
837  pa_pstEPath->ClassID = ltohs(&(pnRunner));
838  i++;
839  break;
840 
841  case 0x24: /* InstanceNr */
842  pa_pstEPath->InstanceNr = *(EIP_UINT8 *) (pnRunner + 1);
843  pnRunner += 2;
844  break;
845 
846  case 0x25: /* InstanceNr 16Bit */
847  pnRunner += 2;
848  pa_pstEPath->InstanceNr = ltohs(&(pnRunner));
849  i++;
850  break;
851 
852  case 0x30: /* AttributeNr */
853  pa_pstEPath->AttributNr = *(EIP_UINT8 *) (pnRunner + 1);
854  pnRunner += 2;
855  break;
856 
857  case 0x31: /* AttributeNr 16Bit */
858  pnRunner += 2;
859  pa_pstEPath->AttributNr = ltohs(&(pnRunner));
860  i++;
861  break;
862 
863  default:
864  OPENER_TRACE_ERR("wrong path requested\n");
865  return EIP_ERROR;
866  break;
867  }
868  }
869 
870  *pa_pnMsg = pnRunner;
871  return i * 2 + 1;
872 }
