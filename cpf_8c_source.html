129 /* INT16 createCPFstructure(INT8 *pa_Data, INT16 pa_DataLength, S_CIP_CPF_Data *pa_CPF_data)
130  * create CPF structure out of pa_data.
131  * pa_Data pointer to data which need to be structured.
132  * pa_DataLength length of data in pa_Data.
133  * pa_CPF_data pointer to structure of CPF data item.
134  * return status
135  * 0 .. success
136  * -1 .. error
137  */
138 EIP_STATUS
139 createCPFstructure(EIP_UINT8 * pa_Data, int pa_DataLength,
140  S_CIP_CPF_Data * pa_CPF_data)
141 {
142  int len_count, i, j;
143 
144  pa_CPF_data->AddrInfo[0].TypeID = 0;
145  pa_CPF_data->AddrInfo[1].TypeID = 0;
146 
147  len_count = 0;
148  pa_CPF_data->ItemCount = ltohs(&pa_Data);
149  len_count += 2;
150  if (pa_CPF_data->ItemCount >= 1)
151  {
152  pa_CPF_data->stAddr_Item.TypeID = ltohs(&pa_Data);
153  pa_CPF_data->stAddr_Item.Length = ltohs(&pa_Data);
154  len_count += 4;
155  if (pa_CPF_data->stAddr_Item.Length >= 4)
156  {
157  pa_CPF_data->stAddr_Item.Data.ConnectionIdentifier = ltohl(&pa_Data);
158  len_count += 4;
159  }
160  if (pa_CPF_data->stAddr_Item.Length == 8)
161  {
162  pa_CPF_data->stAddr_Item.Data.SequenceNumber = ltohl(&pa_Data);
163  len_count += 4;
164  }
165  }
166  if (pa_CPF_data->ItemCount >= 2)
167  {
168  pa_CPF_data->stDataI_Item.TypeID = ltohs(&pa_Data);
169  pa_CPF_data->stDataI_Item.Length = ltohs(&pa_Data);
170  pa_CPF_data->stDataI_Item.Data = pa_Data;
171  pa_Data += pa_CPF_data->stDataI_Item.Length;
172  len_count += (4 + pa_CPF_data->stDataI_Item.Length);
173  }
174  for (j = 0; j < (pa_CPF_data->ItemCount - 2); j++) /* TODO there needs to be a limit check here???*/
175  {
176  pa_CPF_data->AddrInfo[j].TypeID = ltohs(&pa_Data);
177  len_count += 2;
178  if ((pa_CPF_data->AddrInfo[j].TypeID == CIP_ITEM_ID_SOCKADDRINFO_O_TO_T)
179  || (pa_CPF_data->AddrInfo[j].TypeID == CIP_ITEM_ID_SOCKADDRINFO_T_TO_O))
180  {
181  pa_CPF_data->AddrInfo[j].Length = ltohs(&pa_Data);
182  pa_CPF_data->AddrInfo[j].nsin_family = ltohs(&pa_Data);
183  pa_CPF_data->AddrInfo[j].nsin_port = ltohs(&pa_Data);
184  pa_CPF_data->AddrInfo[j].nsin_addr = ltohl(&pa_Data);
185  for (i = 0; i < 8; i++)
186  {
187  pa_CPF_data->AddrInfo[j].nasin_zero[i] = *pa_Data;
188  pa_Data++;
189  }
190  len_count += 18;
191  }
192  else
193  { /* no sockaddr item found */
194  pa_CPF_data->AddrInfo[j].TypeID = 0; /* mark as not set */
195  pa_Data -= 2;
196  }
197  }
198  /* set the addressInfoItems to not set if they werent received */
199  if (pa_CPF_data->ItemCount < 4)
200  {
201  pa_CPF_data->AddrInfo[1].TypeID = 0;
202  if (pa_CPF_data->ItemCount < 3)
203  {
204  pa_CPF_data->AddrInfo[0].TypeID = 0;
205  }
206  }
207  if (len_count == pa_DataLength)
208  { /* length of data is equal to length of Addr and length of Data */
209  return EIP_OK;
210  }
211  else
212  {
213  OPENER_TRACE_WARN("something is wrong with the length in MR @ createCPFstructure\n");
214  if (pa_CPF_data->ItemCount > 2)
215  {
216  /* there is an optional packet in data stream which is not sockaddr item */
217  return EIP_OK;
218  }
219  else
220  { /* something with the length was wrong */
221  return EIP_ERROR;
222  }
223  }
224 }

226 /* INT8 assembleLinearMsg(S_CIP_MR_Response *pa_MRResponse, S_CIP_CPF_Data *pa_CPFDataItem, INT8 *pa_msg)
227  * copy data from MRResponse struct and CPFDataItem into linear memory in pa_msg for transmission over in encapsulation.
228  * pa_MRResponse pointer to message router response which has to be aligned into linear memory.
229  * pa_CPFDataItem pointer to CPF structure which has to be aligned into linear memory.
230  * pa_msg pointer to linear memory.
231  * return length of reply in pa_msg in bytes
232  * -1 .. error
233  */
234 int
235 assembleLinearMsg(S_CIP_MR_Response * pa_MRResponse,
236  S_CIP_CPF_Data * pa_CPFDataItem, EIP_UINT8 * pa_msg)
237 {
238  int i, j, size, type;
239 
240  size = 0;
241  if (pa_MRResponse)
242  {
243  /* add Interface Handle and Timeout = 0 -> only for SendRRData and SendUnitData necessary */
244  htoll(0, &pa_msg);
245  htols(0, &pa_msg);
246  size += 6;
247  }
248 
249  htols(pa_CPFDataItem->ItemCount, &pa_msg); /* item count */
250  size += 2;
251  /* process Address Item */
252  if (pa_CPFDataItem->stAddr_Item.TypeID == CIP_ITEM_ID_NULL)
253  { /* null address item -> address length set to 0 */
254  htols(CIP_ITEM_ID_NULL, &pa_msg);
255  htols(0, &pa_msg);
256  size += 4;
257  }
258  if (pa_CPFDataItem->stAddr_Item.TypeID == CIP_ITEM_ID_CONNECTIONBASED)
259  { /* connected data item -> address length set to 4 and copy ConnectionIdentifier */
260  htols(CIP_ITEM_ID_CONNECTIONBASED, &pa_msg);
261  htols(4, &pa_msg);
262  htoll(pa_CPFDataItem->stAddr_Item.Data.ConnectionIdentifier, &pa_msg);
263  size += 8;
264  }
265  /* sequencenumber????? */
266  if (pa_CPFDataItem->stAddr_Item.TypeID == CIP_ITEM_ID_SEQUENCEDADDRESS)
267  { /* sequenced address item -> address length set to 8 and copy ConnectionIdentifier and SequenceNumber */
268  htols(CIP_ITEM_ID_SEQUENCEDADDRESS, &pa_msg);
269  htols(8, &pa_msg);
270  htoll(pa_CPFDataItem->stAddr_Item.Data.ConnectionIdentifier, &pa_msg);
271  htoll(pa_CPFDataItem->stAddr_Item.Data.SequenceNumber, &pa_msg);
272  size += 12;
273  }
274 
275  /* process Data Item */
276  if ((pa_CPFDataItem->stDataI_Item.TypeID == CIP_ITEM_ID_UNCONNECTEDMESSAGE)
277  || (pa_CPFDataItem->stDataI_Item.TypeID
278  == CIP_ITEM_ID_CONNECTIONTRANSPORTPACKET))
279  {
280  if (pa_MRResponse)
281  {
282  htols(pa_CPFDataItem->stDataI_Item.TypeID, &pa_msg);
283 
284  if (pa_CPFDataItem->stDataI_Item.TypeID
285  == CIP_ITEM_ID_CONNECTIONTRANSPORTPACKET)
286  {
287  htols(
288  (EIP_UINT16) (pa_MRResponse->DataLength + 4 + 2
289  + (2 * pa_MRResponse->SizeofAdditionalStatus)), &pa_msg);
290 
291  htols(
292  (EIP_UINT16) g_stCPFDataItem.stAddr_Item.Data.SequenceNumber,
293  &pa_msg);
294 
295  size += (4 + pa_MRResponse->DataLength + 4 + 2
296  + (2 * pa_MRResponse->SizeofAdditionalStatus));
297  }
298  else
299  {
300  htols(
301  (EIP_UINT16) (pa_MRResponse->DataLength + 4
302  + (2 * pa_MRResponse->SizeofAdditionalStatus)), &pa_msg);
303  size += (4 + pa_MRResponse->DataLength + 4
304  + (2 * pa_MRResponse->SizeofAdditionalStatus));
305  }
306 
307  /* write MR Response into linear memory */
308  *pa_msg = pa_MRResponse->ReplyService;
309  pa_msg++;
310  *pa_msg = pa_MRResponse->Reserved; /* reserved = 0 */
311  pa_msg++;
312  *pa_msg = pa_MRResponse->GeneralStatus;
313  pa_msg++;
314  *pa_msg = pa_MRResponse->SizeofAdditionalStatus;
315  pa_msg++;
316  for (i = 0; i < pa_MRResponse->SizeofAdditionalStatus; i++)
317  htols(pa_MRResponse->AdditionalStatus[i], &pa_msg);
318 
319  for (i = 0; i < pa_MRResponse->DataLength; i++)
320  {
321  *pa_msg = (EIP_UINT8) *(pa_MRResponse->Data + i);
322  pa_msg++;
323  }
324  }
325  else
326  { /* connected IO Message to send */
327  htols(pa_CPFDataItem->stDataI_Item.TypeID, &pa_msg);
328  htols(pa_CPFDataItem->stDataI_Item.Length, &pa_msg);
329  for (i = 0; i < pa_CPFDataItem->stDataI_Item.Length; i++)
330  {
331  *pa_msg = (EIP_UINT8) *(pa_CPFDataItem->stDataI_Item.Data + i);
332  pa_msg++;
333  }
334  size += (pa_CPFDataItem->stDataI_Item.Length + 4);
335  }
336  }
337  /* process SockAddr Info Items */
338  /* make sure first the O->T and then T->O appears on the wire.
339  * EtherNet/IP specification doesn't demand it, but there are EIP
340  * devices which depend on CPF items to appear in the order of their
341  * ID number */
342  for (type = CIP_ITEM_ID_SOCKADDRINFO_O_TO_T; type <= CIP_ITEM_ID_SOCKADDRINFO_T_TO_O; type++)
343  {
344  for (j = 0; j < 2; j++)
345  {
346  if (pa_CPFDataItem->AddrInfo[j].TypeID == type)
347  {
348  htols(pa_CPFDataItem->AddrInfo[j].TypeID, &pa_msg);
349  htols(pa_CPFDataItem->AddrInfo[j].Length, &pa_msg);
350 
351  encapsulateIPAdressCPF(pa_CPFDataItem->AddrInfo[j].nsin_port,
352  pa_CPFDataItem->AddrInfo[j].nsin_addr, pa_msg);
353  pa_msg += 8;
354 
355  memset(pa_msg, 0, 8);
356  pa_msg += 8;
357  size += 20;
358  break;
359  }
360  }
361  }
362  return size;
363 }
